테이블 설계(거의 관례적인 형식) 이 틀에서 뺄거빼면됨
(비인증형 일반 게시판)
create table board(
	idx number constraint pk_board_idx primary key,
    --글번호 (오라클 : sequence 객체) --MS-sql , mysql(테이블에 종속 자동증가)
    writer varchar2(30) not null,
    --글쓴이(필수 입력 사항 : 회원전용 > 로그인한 ID값 (session) , 비회원전용 : 입력)
    pwd varchar2(30) not null,
    --회원전용(x) , 비회원(0)
    subject varchar2(50) not null, --글제목
    content varchar2(2000) not null, 글내용
    filename varchar2(50), --파일명 또는 파일경로 (note.hwp) : 실제파일은 웹서버에 관리
    filesize number, --byte 옵션
    writedate date default SYSDATE, --기본값 설정 (DB 서버 날짜 insert)
    readnum number default 0, --글 조회수 증가 (업무 로직 구현)
    homepage varchar2(20), --옵션 ...
    email varchar2(20) -- 옵션 ... 필수 입력사항이 아님 (null 허용) 
);

(비인증형 계층형 게시판) : 이것만 하면 끝남(어려움)
create table multiboard(
	idx number constraint pk_board_idx primary key,
    --글번호 (오라클 : sequence 객체) --MS-sql , mysql(테이블에 종속 자동증가)
    writer varchar2(30) not null,
    --글쓴이(필수 입력 사항 : 회원전용 > 로그인한 ID값 (session) , 비회원전용 : 입력)
    pwd varchar2(30) not null,
    --회원전용(x) , 비회원(0)
    subject varchar2(50) not null, --글제목
    content varchar2(2000) not null, 글내용
    filename varchar2(50), --파일명 또는 파일경로 (note.hwp)
    filesize number, --byte 옵션
    writedate date default SYSDATE, --기본값 설정 (DB 서버 날짜 insert)
    readnum number default 0, --글 조회수 증가 (업무 로직 구현)
    homepage varchar2(20), --옵션 ...
    email varchar2(20), -- 옵션 ... 필수 입력사항이 아님 (null 허용) 
    
    --추가적인 컬럼(★ key point)
    refer number default 0, --참조글 or 글의 그룹번호 / ref : 글의 묶음
    depth number default 0, --글의 깊이, 들여쓰기 (화면처리) 로직 (x)
    step  number default 0, --글의 정렬 순서 , 답글 정렬
);

idx ,  ref  , depth , step ,   		title
1       1       0      0       		원본글  
2       2       0      0       		원본글2
3       1       1      1       		원본글     답글1-1 (원본글 : idx:1)
4       2       1      1       		원본글2  답글2-1 (원본글 : idx:2) 
5       1       2      2(+1)> 3     답글1-2  답글1-1 (읽고 있는 글 : idx:3) 3번에 대한 답글★★★★★
6       1       2      2        	답글1-1  답글1-1 (읽고 있는 글 : idx:3)  3번에 대한 답글 ★★★★★ step은 중복 XX
7		1		2	   2(2를 쓸라면 6번과 5번 1씩 더함)	

--------------------
order by ref desc  , step asc : 계층형 게시판의 첫번재 쿼리, ref로 디센딩(의역 : 최신글을 앞에 두겠다)
								step asc : 나중에쓴 답글을 위로 올리겠다?

depth가 없으면 들여쓰기가 안됨(로직상 문제는 없음)
step은 반드시 있어야됨

★네이버는 다름(step 부분)
우리는 최신 답글을 위로 올림
네이버는 최신답글을 뒤로 보냄
이 논리는 네이버꺼쓰거나 안쓰거나는 설계자마음대로

화면에 출력
idx

>4          2    1    1 
1           1    0    0
>3          1    1    1
>>7         1    2    2
>>6         1    2    2+1 > 3
>>5         1    2    3+1 > 4

계층형 게시판
논리 2가지
1. 최신 답글을 위로
2. 답글쓴 순서대로 

1. 최신 답글을 위로
규칙 : 원본글 : depth , step >> 0 0
	 (같은  ref)현재 read 한 글의 step 값다 더 큰 step 값이 있으면 +1  
          답글 : depth+1  step+1

select * from multiboard order by ref desc , step asc
